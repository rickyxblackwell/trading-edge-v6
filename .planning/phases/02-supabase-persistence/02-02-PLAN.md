```typescript
"use client"

import { useState, useCallback, useRef } from "react"

export function useToast() {
  const [toast, setToast] = useState<string | null>(null)
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null)

  const showToast = useCallback((message: string) => {
    if (timerRef.current) clearTimeout(timerRef.current)
    setToast(message)
    timerRef.current = setTimeout(() => setToast(null), 4000)
  }, [])

  return { toast, showToast }
}
```

Notes:
- `useRef` for the timer handle — ensures successive `showToast` calls cancel the in-flight dismissal (avoids the "old toast disappears mid-display of new toast" glitch).
- `useCallback` with empty deps array — `setToast` is stable, no closure dependencies needed.
- 4000ms matches the Account.tsx 5000ms confirm-timeout pattern but slightly shorter for transient errors (consistent with RESEARCH.md Pattern 7).
- No transition/animation here — that lives in Toast.tsx.
